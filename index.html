Code example

Find the sum of all multiples of n below m.


function sumMul(n, m) {
    let sum = 0;
    let summand = n;
                                        
    if (n >= m) {
        return "INVALID";
    }
    
    while (summand < m) {
        sum += summand;
        summand += n;
    }
    return sum;
}
